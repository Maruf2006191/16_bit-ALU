module alu_clock_gated #(
    parameter WIDTH = 16
)(
    input  logic             clk,
    input  logic             rst_n,
    input  logic             enable,
    input  logic [WIDTH-1:0] A,
    input  logic [WIDTH-1:0] B,
    input  logic [3:0]       opcode,
    output logic [WIDTH-1:0] result,
    output logic             zero_flag,
    output logic             overflow_flag
);

    // Gated operand registers
    logic [WIDTH-1:0] gated_operandA;
    logic [WIDTH-1:0] gated_operandB;
    logic [3:0]       gated_opcode;
    
    // Internal signals
    logic [WIDTH-1:0] next_result;
    logic             next_zero;
    logic             next_overflow;
    logic [WIDTH:0]   temp_add_sub;

    // Latch inputs only when enabled to reduce toggling
    always_ff @(posedge clk or negedge rst_n) begin
        if (!rst_n) begin
            gated_operandA <= '0;
            gated_operandB <= '0;
            gated_opcode   <= '0;
        end else if (enable) begin
            gated_operandA <= A;
            gated_operandB <= B;
            gated_opcode   <= opcode;
        end
    end

    // Combinational logic uses gated operands
    always_comb begin
        next_result   = '0;
        next_overflow = 1'b0;
        temp_add_sub  = '0;

        case (gated_opcode)
            4'b0000: begin
                temp_add_sub = gated_operandA + gated_operandB;
                next_result  = temp_add_sub[WIDTH-1:0];
                next_overflow = (gated_operandA[WIDTH-1] == gated_operandB[WIDTH-1]) && 
                                (next_result[WIDTH-1] != gated_operandA[WIDTH-1]);
            end
            4'b0001: begin
                temp_add_sub = gated_operandA - gated_operandB;
                next_result  = temp_add_sub[WIDTH-1:0];
                next_overflow = (gated_operandA[WIDTH-1] != gated_operandB[WIDTH-1]) && 
                                (next_result[WIDTH-1] != gated_operandA[WIDTH-1]);
            end
            4'b0010: next_result = gated_operandA * gated_operandB; 
            4'b0011: next_result = (gated_operandB != 0) ? (gated_operandA / gated_operandB) : '0; 
            4'b0100: next_result = gated_operandA & gated_operandB; 
            4'b0101: next_result = gated_operandA | gated_operandB; 
            4'b0110: next_result = gated_operandA ^ gated_operandB; 
            4'b0111: next_result = ~gated_operandA;    
            4'b1000: next_result = gated_operandA << 1; 
            4'b1001: next_result = gated_operandA >> 1; 
            default: next_result = '0;
        endcase

        next_zero = (next_result == '0) ? 1'b1 : 1'b0;
    end

    // Update results only when enable is high
    always_ff @(posedge clk or negedge rst_n) begin
        if (!rst_n) begin
            result        <= '0;
            zero_flag     <= 1'b0;
            overflow_flag <= 1'b0;
        end else if (enable) begin
            result        <= next_result;
            zero_flag     <= next_zero;
            overflow_flag <= next_overflow;
        end
    end

endmodule
